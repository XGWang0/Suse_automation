  91. /**
  92.   * Searches waiver tables.
  93.   * $mode:
  94.   *   0 waiver_data
  95.   *   1 waiver_data + waiver_testcase
  96.   *   2 waiver_testcase only
  97.   * $attrs['only_id']:
  98.   *   0 for full details
  99.   *   1 for IDs only
  100.   * $attrs['order_nr']: +/- (1+<index of the result column>)
  101.   *   positive nr. means ascending order, negative means descending.
  102.   *
  103.   * NOTE: see search_submissions_results for similar code & more comments
  104.   * @return array 2D array of results
  105.   ***/
  106. function search_waiver($mode,$attrs)
  107. {
  108.     # supported arguments
  109.     $attrs_known=array(
  110.         'waiverID'=> array('d.waiverID=?','i'),
  111.         'waiver_tcID'=>array('t.waiver_tcID=?','i'),
  112.         'bugID'=>array('d.bugID=?','i'),
  113.         'explanation'=>array('d.explanation like ?','s'),
  114.         'productID'=>array('t.productID=?','i'),
  115.         'releaseID'=>array('t.releaseID=?','i'),
  116.         'testcaseID'=>array('d.testcaseID=?','i'),
  117.         'archID'=>array('t.archID=?','i'),
  118.         'matchtype'=>array('t.matchtype=?','i')
  119.     );
  120.     $only_id=hash_get($attrs,'only_id',false,true);
  121.     # $sel0[ $mode ] - base attribute
  122.     $sel0=array( 'd.waiverID', 't.waiver_tcID', 't.waiver_tcID' );
  123.     # $sel1[ $mode ] - additional attributes for (!$only_id)
  124.     $sel1=array(
  125.         array('d.testcaseID','d.bugID','d.explanation'),
  126.         array('d.testcaseID','d.waiverID','t.productID','t.releaseID','t.archID','t.matchtype'),
  127.         array('t.productID','t.releaseID','t.archID','t.matchtype')
  128.     );
  129.     # $from0[ $mode ] - tables to select from
  130.     $from0=array('waiver_data d','waiver_data d JOIN waiver_testcase t USING(waiverID)');
  131.     $from0[2]=$from0[1];
  132.     $sel=$sel0[$mode];
  133.     if( !$only_id )
  134.         $sel=array_merge( array($sel), $sel1[$mode] );
  135.     return search_common( $sel, $from0[$mode], $attrs, $attrs_known );
  136. }

  193. function search_submissions_results($mode, $attrs)
  194. {
  195.     # supported arguments
  196.     $attrs_known=array(
  197.         'date_from'    => array('s.submission_date>=?',    's'),
  198.         'date_to'    => array('s.submission_date<=?',    's'),
  199.         'hostID'    => array('s.hostID=?',            'i'),
  200.         'testerID'    => array('s.testerID=?',        'i'),
  201.         'archID'    => array('s.archID=?',            'i'),
  202.         'productID'    => array('s.productID=?',        'i'),
  203.         'releaseID'    => array('s.releaseID=?',        'i'),
  204.         'active'    => array('s.active=?',            'i'),
  205.         'testsuiteID'   => array('g.testsuiteID=?',        'i'),
  206.         'testcaseID'    => array('r.testcaseID=?',        'i'),
  207.         'testcase'    => array('c.testcase like ?',    's'),
  208.         'tcfID'        => array('r.tcfID=?',            'i'),
  209.         'submissionID'    => array('s.submissionID=?',        'i'),
  210.         'configID'    => array('s.configID=?',        'i'),
  211.         'comment'    => array('s.comment like ?',        's'),
  212.     );
  213.     # $what: 0: submissions. 1: results, 2: testsuites (TODO: ughly, rewrite)
  214.     $i_main = array( 1,0,0,0,0,0,2,0,0,0,0 );
  215.     $i_next = array( 0,0,1,2,0,3,0,4,5,5,6 );
  216.     $i_from = array( 0,0,1,2,3,4,0,0,5,6,7 );
  217. #    $what     = ($mode==0 ? 1 : ($mode==6 ? 2:0));
  218. #    if($what!=0) 
  219. #        $mode=0;
  220. #    else
  221. #        $mode--;
  222.     $only_id  = hash_get($attrs,'only_id',false,true);
  223.  
  224.     # SELECT $sel FROM $from
  225.     # $sel = $sel0 [ $sel1 ] $sel2
  226.     # $from = $from0 $from2
  227.     # $sel0..$sel2 are done as lists - for ordering by them
  228.     # $sel0[ $i_main ] -- always
  229.     $sel0=array( 's.submissionID', 'r.resultsID', 'g.testsuiteID' );
  230.     # $sel1[ $i_main ] -- appends for full details
  231.     $sel1=array( 
  232. /* subms */  array('s.submission_date','s.hostID','s.testerID','s.archID','s.productID','s.releaseID','s.active','s.related','s.comment','s.configID'),
  233. /* rslts */  array('g.testsuiteID','r.testcaseID','r.succeeded','r.failed','r.internal_error','r.times_run','r.test_time'),
  234. /* suite */  array()
  235.     );
  236.     # $sel2[ $i_next ] -- appends for full details
  237.     $sel2=array( 
  238. /* simple */ array(),
  239. /* mtnce  */ array('m.patchID','m.md5sum','m.status'),
  240. /* KOTD   */ array('k.release','k.version','k.branchID'),
  241. /* any    */ array('m.patchID','m.md5sum','m.status'),
  242. /* trend  */ array('g.testsuiteID'),
  243. /* TCF    */ array('g.testsuiteID','g.tcfID'),
  244. /* TCF+res*/ array('g.testsuiteID','g.tcfID','r.testcaseID'),
  245.     );
  246.  
  247.     # $from0[ $i_main ] -- always
  248.     $from0=array( 'submissions s', 'submissions s JOIN tcf_group g USING(submissionID) JOIN results r USING(tcfID)', 'submissions s JOIN tcf_group g USING(submissionID)' );
  249.     # $from2[ $i_from ] -- always
  250.     $from2=array(
  251. /* simple */    '',
  252. /* mtnce  */    '  JOIN maintenance_testing m USING(submissionID)',
  253. /* KOTD   */    ' JOIN kotd_testing k USING(submissionID)',
  254. /* prod   */    ' JOIN product_testing p USING(submissionID)',
  255. /* any    */    ' LEFT JOIN kotd_testing k USING(submissionID) LEFT JOIN maintenance_testing m USING(submissionID) LEFT JOIN product_testing p USING(submissionID)',
  256. /* TCF    */    ' JOIN tcf_group g USING(submissionID)',
  257. /* bench  */    ' JOIN tcf_group g USING(submissionID) JOIN bench_suites b USING(testsuiteID)',
  258. /* TCF+res*/    ' JOIN tcf_group g USING(submissionID) JOIN results r USING(tcfID) JOIN testcases c USING(testcaseID)',
  259.     );
  260.     
  261.     # make the SQL base SELECT
  262.     $sel=array($sel0[$i_main[$mode]]);
  263.     if( !$only_id )
  264.         $sel=array_merge($sel, $sel1[$i_main[$mode]], $sel2[$i_next[$mode]]);
  265.     $from=$from0[$i_main[$mode]].$from2[$i_from[$mode]];
  266.     return search_common( $sel, $from, $attrs, $attrs_known  );
  267. }
