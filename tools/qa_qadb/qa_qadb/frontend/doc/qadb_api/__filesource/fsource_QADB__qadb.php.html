  94. /**
  95.   * Searches waiver tables.
  96.   * $mode:
  97.   *   0 waiver
  98.   *   1 waiver + waiver_testcase
  99.   *   2 waiver_testcase only
  100.   * $attrs['only_id']:
  101.   *   0 for full details
  102.   *   1 for IDs only
  103.   * $attrs['order_nr']: +/- (1+<index of the result column>)
  104.   *   positive nr. means ascending order, negative means descending.
  105.   *
  106.   * NOTE: see search_submission_result for similar code & more comments
  107.   * @return array 2D array of result
  108.   ***/
  109. function search_waiver($mode,$attrs)
  110. {
  111.     # supported arguments
  112.     $attrs_known=array(
  113.         'waiver_id'=> array('d.waiver_id=?','i'),
  114.         'waiver_testcase_id'=>array('t.waiver_testcase_id=?','i'),
  115.         'bug_id'=>array('d.bug_id=?','i'),
  116.         'explanation'=>array('d.explanation like ?','s'),
  117.         'product_id'=>array('t.product_id=?','i'),
  118.         'release_id'=>array('t.release_id=?','i'),
  119.         'testcase_id'=>array('d.testcase_id=?','i'),
  120.         'arch_id'=>array('t.arch_id=?','i'),
  121.         'matchtype'=>array('t.matchtype=?','i')
  122.     );
  123.     $only_id=hash_get($attrs,'only_id',false,true);
  124.     # $sel0[ $mode ] - base attribute
  125.     $sel0=array( 'd.waiver_id', 't.waiver_testcase_id', 't.waiver_testcase_id' );
  126.     # $sel1[ $mode ] - additional attributes for (!$only_id)
  127.     $sel1=array(
  128.         array('d.testcase_id','d.bug_id','d.explanation'),
  129.         array('d.testcase_id','d.waiver_id','t.product_id','t.release_id','t.arch_id','t.matchtype'),
  130.         array('t.product_id','t.release_id','t.arch_id','t.matchtype')
  131.     );
  132.     # $from0[ $mode ] - tables to select from
  133.     $from0=array('waiver d','waiver d JOIN waiver_testcase t USING(waiver_id)');
  134.     $from0[2]=$from0[1];
  135.     $sel=$sel0[$mode];
  136.     if( !$only_id )
  137.         $sel=array_merge( array($sel), $sel1[$mode] );
  138.     return search_common( $sel, $from0[$mode], $attrs, $attrs_known );
  139. }

  384. function regression_differences($attrs,&$pager=null)
  385. {
  386.     # This is ugly, but comparing submission of 100k testcase was no longer possible in PHP
  387.     # What happens here: 
  388.     # we link candidate and reference data using testcase_id
  389.     # on-the-fly we compute status succ(s), fail(f), interr(i) or skipped(S) for both
  390.     # we filter out testcase with the same status
  391.     #
  392.     # As the status is auto-generated, MySQL seems only able to filter it in an outside SELECT
  393.     # We use search_common() to select proper data and generate SQL
  394.     # then we add the outer SELECT with filtering out testcase with the same status
  395.     
  396.     # ugly hack: the first two columns are separated so that we can order by index 2, the others not
  397.     $sel=array("c.testcase_id","testcase",
  398.         "GROUP_CONCAT(CASE WHEN tr.log_url IS NULL THEN sr.testsuite ELSE CONCAT('<a href=\"',tr.log_url,'/',relative_url,'\">',sr.testsuite,'</a>') END) AS r_testsuite, 
  399.             SUM(r.succeeded) AS r_succ, 
  400.             SUM(r.failed) AS r_fail, 
  401.             SUM(r.internal_error) AS r_interr, 
  402.             SUM(r.skipped) AS r_skip, 
  403.             (CASE WHEN r.failed>0 THEN 'f' WHEN r.internal_error>0 THEN 'i' WHEN r.skipped>0 THEN 'S' WHEN r.succeeded>0 THEN 's' ELSE 'i' END) AS r_state,
  404.          GROUP_CONCAT(CASE WHEN tc.log_url IS NULL THEN sc.testsuite ELSE CONCAT('<a href=\"',tc.log_url,'/',relative_url,'\">',sc.testsuite,'</a>') END) AS c_testsuite, 
  405.              SUM(c.succeeded) AS c_succ, 
  406.              SUM(c.failed) AS c_fail, 
  407.             SUM(c.internal_error) AS c_interr, 
  408.             SUM(c.skipped) AS c_skip, 
  409.             (CASE WHEN c.failed>0 THEN 'f' WHEN c.internal_error>0 THEN 'i' WHEN c.skipped>0 THEN 'S' WHEN c.succeeded>0 THEN 's' ELSE 'i' END) AS c_state,
  410.         waiver_id
  411.     ");
  412.     $from="result c JOIN tcf_group tc USING(tcf_id) JOIN testsuite sc USING(testsuite_id) JOIN testcase ss USING(testcase_id), result r JOIN tcf_group tr USING(tcf_id) JOIN testsuite sr USING(testsuite_id) LEFT OUTER JOIN waiver USING(testcase_id)";
  413.     $where="c.testcase_id=r.testcase_id";
  414.  
  415.     # attributes
  416.     $attrs_known=array(
  417.         'cand_submission_id'    => array('tc.submission_id=?','i'),
  418.         'ref_submission_id'    => array('tr.submission_id=?','i'),
  419.         'cand_tcf_id'        => array('c.tcf_id=?','i'),
  420.         'ref_tcf_id'        => array('r.tcf_id=?','i')
  421.     );
  422.  
  423.     # check that both cand/ref are specified, would block DB otherwise
  424.     if(!(    (isset($attrs['cand_submission_id']) || isset($attrs['cand_tcf_id'])) &&
  425.         (isset($attrs['ref_submission_id'])  || isset($attrs['ref_tcf_id'] )) ))
  426.         die("Called regression_difference() without specifying cand/ref, would block DB");
  427.  
  428.     # other query settings
  429.     $attrs['just_sql']=true;
  430.     $attrs['where']=$where;
  431.     $attrs['order_nr']=2;
  432.     $attrs['group_by']='c.testcase_id';
  433.     
  434.     # disable inner select's default limit of search_common()
  435.     if( !isset($attrs['limit']) )
  436.         $attrs['limit']=array();    
  437.  
  438.     # generate inner query
  439.     $args=search_common($sel,$from,$attrs,$attrs_known);
  440.     $sql="FROM (".$args[2].") tbl WHERE c_state!=r_state";
  441.  
  442.     # paging
  443.     $limit=null;
  444.     if( !is_null($pager) )    {
  445.         $cargs = $args;
  446.         $cargs[2] = "SELECT COUNT(*) $sql";
  447.         array_splice($cargs,0,2);
  448.         $count = call_user_func_array('scalar_query',$cargs);
  449.         $limit = limit_from_pager($pager,$count);
  450.     }
  451.  
  452.     # get data
  453.     $args[2] = "SELECT * $sql".(is_null($limit) ? '' : " LIMIT ".$limit[0].','.$limit[1]);
  454.  
  455.     return call_user_func_array('mhash_query',$args);
  456. }
