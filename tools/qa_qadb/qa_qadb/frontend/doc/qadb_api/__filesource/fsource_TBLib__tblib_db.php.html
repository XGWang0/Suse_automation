  16. /**
  17.   * Common search code.
  18.   * Builds a search query, executes, returns the results.
  19.   * @param array $sel array of attributes to select
  20.   * @param string $from string containing the join to search from
  21.   * @param array &$attrs requested search values
  22.   * @param array &$attrs_known array of array(<SQL condition>, <bind format letter>)
  23.   * @return array 2D array with search results
  24.   ***/
  25. function search_common( $sel, $from, &$attrs, &$attrs_known )
  26. {
  27.     $header   = hash_get($attrs,'header',true,true);
  28.     $limit    = hash_get($attrs,'limit',array(5000),true);
  29.     $order_nr = hash_get($attrs,'order_nr',0,true);
  30.  
  31.     $select=join(', ',$sel);
  32.     $sql="SELECT $select FROM $from WHERE 1";
  33.     $order='';
  34.     if( $order_nr )
  35.         $order=' ORDER BY '.$sel[abs($order_nr)-1].($order_nr>0 ? ' ASC':' DESC');
  36.  
  37.     
  38.     $args=array();
  39.     $format='';
  40.  
  41.     foreach($attrs as $key=>$val)
  42.     {
  43.         # check correct calling
  44.         if( !isset($attrs_known[$key]) )
  45.             abort("search_common: unknown attribute : $key");
  46.         if( !set($val) ) continue;
  47.         if( is_array($val) && count($val)==0 )    continue;
  48.  
  49.         $a=$attrs_known[$key];
  50.         
  51.         # sanitize one-value arrays
  52.         if( is_array($val) && count($val)==1 )
  53.             $val=$val[0];
  54.  
  55.         if( !is_array($val)  )
  56.         {    # one value
  57.             $sql    .= ' AND '.$a[0];
  58.             $format .= $a[1];
  59.             $args[] .= $val;
  60.         }
  61.         else
  62.         {    # multiple values
  63.             $sql .= ' AND ( 0';
  64.             for( $i=0; $i<count($val); $i++ )
  65.             {
  66.                 $sql    .= ' OR '.$a[0];
  67.                 $format    .= $a[1];
  68.                 $args[]    .= $val[$i];
  69.             }
  70.             $sql .= ')';
  71.         }
  72.     }
  73.     $sql.=$order;
  74. #    print("SQL : $sql<br/>\n");
  75. #    print("format : $format<br/>\n");
  76. #    print("Arguments : ");
  77. #    print_r($args);
  78.     array_unshift($args, $header, $limit, $sql, $format);
  79.     return call_user_func_array('matrix_query', $args);
  80. }

  426. /**

  433. function cache_statement($query)
  434. {
  435.     global $mysqli,$sql_cache,$sql_cache_size,$sql_cache_hits,$sql_cache_misses,$sql_cache_replaces;
  436.     if( ! array_key_exists($query,$sql_cache) )

  439.         if( count($sql_cache) >= $sql_cache_size )
  440.         {
  441.             mysqli_stmt_close(array_shift($sql_cache));
  442.             $sql_cache_replaces++;
  443.         }
  444.         $statement=$mysqli->prepare($query);
  445.         if(!$statement) abort("Cannot prepare statement '$query': ".$mysqli->error);
  446.         $sql_cache[$query]=$statement;
  447.     }
  448.     else
  449.     {
  450.         $sql_cache_hits++;
  451.         $statement=$sql_cache[$query];
  452.         unset($sql_cache[$query]);

  467.     global $maxqsql,$maxqtime;
  468.     $start=time();
  469.     $args=func_get_args();
  470.     $limit=array_shift($args);
  471.     $query=array_shift($args);
  472.     if($limit && count($limit)<3)
  473.     {    // adding the LIMIT clause
  474.         if(!isset($args[0])) $args[0]='';
  475.         $args[0].=str_repeat('i',count($limit));
  476.         $query.=' LIMIT '.join(',',array_fill(0,count($limit),'?'));
  477.         $args=array_merge($args,$limit);
  478.     }
  479.     $statement=cache_statement($query);
  480.     if(count($args)>1)
  481.         call_user_func_array(array($statement,'bind_param'),$args);
  482.     $statement->execute() or abort( $statement->error );
  483.     $statement->store_result();
  484.     $time=time()-$start;
  485.     if($time>$maxqtime)
  486.     {
  487.         $maxqtime=$time;
  488.         $maxqsql=join('|',$args);
  489.         $maxqsql="$query $maxqsql";
  490.     }
  491.     return $statement;
  492. }
  493.

  495.   * Prints HTML info about the SQL statement cache.

  499. function print_cache_info()
  500. {
  501.     global $sql_cache,$sql_cache_size,$sql_cache_hits,$sql_cache_misses,$sql_cache_replaces,$starttime,$maxqtime,$maxqsql;
  502.     $time=time()-$starttime;
  503.     $show=array('max size',$sql_cache_size,'act size',count($sql_cache),'hits',$sql_cache_hits,'misses',$sql_cache_misses,'replaces',$sql_cache_replaces, 'runtime', date('i:s',$time), 'slowest query time', date('i:s',$maxqtime), 'slowest query', $maxqsql);
  504.     print '<div class="sqlinfo"><span class="header">SQL cache info</span>'."\n";
  505.     for( $i=0; $i<count($show); $i+=2 )

  508. }
